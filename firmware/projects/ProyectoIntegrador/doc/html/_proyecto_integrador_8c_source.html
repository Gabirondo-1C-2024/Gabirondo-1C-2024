106static void tarea_leer_gas(void *pvParameter)
107{
108 uint16_t vect[5];
109 uint32_t acum = 0;
110 int i = 0;
111
112 while (1)
113 {
114 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); /* La tarea espera en este punto hasta recibir una notificación */
115 if (estadoMedicion)
116 {
117 float read = MQRead();
118 valorGasLPG = MQGetPercentage(read / RO_);
119 valor_convertido = (uint16_t)(valorGasLPG * 1000000);
120 if (i < 5)
121 {
122 vect[i] = valor_convertido;
123 acum += vect[i];
124 i++;
125 }
126 else
127 {
128 promedio = acum / 5;
129 acum = 0;
130 i = 0;
131 }
132
133 // printf("Valor concetracion: %d, %f \n", valor_convertido, read);
